간편 설정 방법:
1. 아래 Google Apps Script 코드를 복사합니다
2. Google Apps Script 에서 새 프로젝트 생성
3. 코드를 붙여넣고 저장합니다
4. "배포" > "새 배포" > "웹 앱" 선택
5. "실행자"를 "나"로, "액세스 권한"을 "모든 사용자"로 설정 후 배포
6. 생성된 웹 앱 URL 을 아래에 입력합니다 

// Google Apps Script 코드 (수정됨 - 2026.04.30 v2)
// JSON 바디를 처리하는 방식 - POST 와 GET 모두 지원

function doGet(e) {
  try {
    var spreadsheetId = e.parameter.spreadsheetId;
    if (!spreadsheetId) {
      return createJsonResponse({
        result: 'error',
        message: 'spreadsheetId 가 필요합니다.'
      });
    }
    
    var ss = SpreadsheetApp.openById(spreadsheetId);
    var sheet = ss.getSheetByName('Data');
    
    if (!sheet) {
      sheet = ss.insertSheet('Data');
    }
    
    var value = sheet.getRange('A1').getValue();
    
    if (e.parameter.action === 'load') {
      var data = value ? JSON.parse(value) : {};
      return createJsonResponse({
        result: 'success',
        data: data
      });
    }
    
    return createJsonResponse({
      result: 'error',
      message: 'Invalid action'
    });
  } catch (error) {
    return createJsonResponse({
      result: 'error',
      message: 'Error: ' + error.toString()
    });
  }
}

function doPost(e) {
  try {
    // JSON 바디에서 데이터 파싱
    var postData = JSON.parse(e.postData.contents);
    var spreadsheetId = postData.spreadsheetId;
    var action = postData.action;
    var data = postData.data;
    
    if (!spreadsheetId) {
      return createJsonResponse({
        result: 'error',
        message: 'spreadsheetId 가 필요합니다.'
      });
    }
    
    var ss = SpreadsheetApp.openById(spreadsheetId);
    var sheet = ss.getSheetByName('Data');
    
    if (!sheet) {
      sheet = ss.insertSheet('Data');
    }
    
    if (action === 'save') {
      sheet.clearContents();
      sheet.getRange('A1').setValue(JSON.stringify(data));
      return createJsonResponse({
        result: 'success',
        message: 'Data saved successfully'
      });
    }
    
    return createJsonResponse({
      result: 'error',
      message: 'Invalid action'
    });
  } catch (error) {
    return createJsonResponse({
      result: 'error',
      message: 'Error: ' + error.toString()
    });
  }
}

function createJsonResponse(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
